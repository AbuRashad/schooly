36def anonymize_minor_faces(
37 frame: np.ndarray,
38 age_estimator: Callable[[np.ndarray], int] | None = None,
39 blur_kernel: tuple[int, int] = (51, 51),
40 detector: cv2.CascadeClassifier | None = None,
41 minor_age_threshold: int = POLICY.minor_age_threshold,
42) -> np.ndarray:
43 """Blur detected minor faces — uses conservative policy if no age estimator."""
44 if frame is None or frame.size == 0:
45 raise ValueError("Input frame must be a non-empty image array.")
46 detector = detector or _load_default_face_detector()
47 gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
48 faces = detector.detectMultiScale(gray, scaleFactor=1.1, minNeighbors=5, minSize=(30,30))
49 anonymized = frame.copy()
50 for (x, y, w, h) in faces:
51 face_region = anonymized[y:y+h, x:x+w]
52 estimated_age = age_estimator(face_region) if age_estimator else 0
53 if estimated_age < minor_age_threshold:
54 anonymized[y:y+h, x:x+w] = cv2.GaussianBlur(face_region, blur_kernel, 0)
55 return anonymized

57def enforce_video_log_ttl(logs_path: str | Path, ttl_hours: int = POLICY.default_video_ttl_hours,
58 now: datetime | None = None, allowed_suffixes: Iterable[str] = (".mp4",".avi",".mov")) -> list[Path]:
59 """Delete expired video files according to strict TTL policy (72h default)."""
60 if ttl_hours <= 0: raise ValueError("ttl_hours must be greater than zero.")
61 base_path = Path(logs_path)
62 if not base_path.exists(): return []
63 current_time = now or datetime.now(timezone.utc)
64 expiry_delta = timedelta(hours=ttl_hours)
65 deleted_files: list[Path] = []
66 for file_path in base_path.rglob("*"):
67 if not file_path.is_file(): continue
68 modified_time = datetime.fromtimestamp(file_path.stat().st_mtime, tz=timezone.utc)
69 if current_time - modified_time > expiry_delta:
70 file_path.unlink(missing_ok=True)
71 deleted_files.append(file_path)
72 return deleted_files

271 def compute_daily_baselines(self) -> dict[str, dict[str, float]]:
272 """Recompute per-student daily engagement averages. Called nightly."""
273 baselines: dict[str, dict[str, list[float]]] = defaultdict(lambda: defaultdict(list))
274 with self._lock:
275 for snap in self._snapshots:
276 date_key = snap.recorded_at.date().isoformat()
277 baselines[snap.student_id][date_key].append(snap.score)
278 result: dict[str, dict[str, float]] = {}
279 for student_id, days_data in baselines.items():
280 result[student_id] = {
281 day: round(sum(scores) / len(scores), 4)
282 for day, scores in days_data.items()}
283 with self._lock: self._baselines = result
